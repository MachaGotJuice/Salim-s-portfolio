function u = Euler(u0 , Nh , T, x)
h = T/Nh ;
u = zeros(2,Nh+1);
u(:,1) = u0;
 
for i = 1:Nh
u(:,i+1) = u(:,i) + h*f(u(:,i), x);
end
end
 
function up = f(u, x)
k = 50;
xe = 0.5;
We = k*(1-xe)*u(2)/(u(1)+u(2));
W = k*(1-x)*u(1)/(u(1)+u(2));
D = 30;
up =[(W-D)*u(1);
(We-D)*u(2)];
end
